```package main

import "log"

func main(){
	supposeNum := 213130219832190
	print(isFibo(0,1,supposeNum))
}

func isFibo(current int,current2 int,dest int)bool{
	if dest < current2{
		return false
	}
	sum := current+current2
	log.Printf("calculating %d",sum)
	if dest == sum{
		return true
	}
	return isFibo(current2,sum,dest)
}
```
